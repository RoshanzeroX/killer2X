// ==UserScript==
// @name          Ultra Auto Resume Videos for MissAV (all domains)
// @namespace     http://tampermonkey.net/
// @version       1.1
// @description   Automatically resume all videos and bypass the "Confirm" screen on the specified domain.
// @match         https://lmsrv.siammakro.co.th/
// @grant         none
// ==/UserScript==

(function () {
    'use strict';

    // Check for the "Confirm" button every second and click it
    const autoClickConfirm = () => {
        const confirmButton = Array.from(document.querySelectorAll('button, div')).find(el => 
            el.textContent.includes('Confirm')
        );

        if (confirmButton && confirmButton.offsetParent !== null) {
            console.log('✅ Found and clicking the Confirm button...');
            confirmButton.click();
        }
    };

    setInterval(autoClickConfirm, 1000);

    // Automatically resume any paused video
    const handledVideos = new WeakSet();

    setInterval(() => {
        const videos = document.querySelectorAll('video');
        videos.forEach(video => {
            if (handledVideos.has(video)) return;
            handledVideos.add(video);

            setInterval(() => {
                if (video.paused && video.readyState >= 2) {
                    video.play().catch(e => console.warn('Play error:', e));
                }
            }, 1000);
        });
    }, 1000);
})();
