// A simple object to hold our shared state
const videoState = {
    currentSpeed: 1.0,
    isMuted: false,
    timeAdvanceFactor: 0
};

// A master switch for the script
let isScriptEnabled = true;

// 1. Function to create the control panel
function createVideoControls() {
    if (document.getElementById('video-control-panel')) {
        console.log('Control panel already exists.');
        return;
    }

    const speedOptions = [1.0, 1.5, 2.0, 3.0, 4.0];
    const panel = document.createElement('div');
    panel.id = 'video-control-panel';
    Object.assign(panel.style, {
        position: 'fixed',
        top: '10px',
        right: '10px',
        zIndex: 99999,
        backgroundColor: 'rgba(0, 0, 0, 0.7)',
        padding: '10px',
        borderRadius: '8px',
        display: 'flex',
        flexDirection: 'column',
        gap: '8px',
        fontFamily: 'Arial, sans-serif',
        color: '#fff'
    });
    
    // Add title
    const title = document.createElement('div');
    title.innerText = 'üé• Video Controls';
    title.style.fontWeight = 'bold';
    panel.appendChild(title);

    // Add mute button
    const muteBtn = document.createElement('button');
    muteBtn.innerText = 'üîä ‡∏õ‡∏¥‡∏î‡πÄ‡∏™‡∏µ‡∏¢‡∏á';
    muteBtn.style.background = '#dc3545';
    Object.assign(muteBtn.style, { padding: '6px 10px', border: 'none', borderRadius: '4px', cursor: 'pointer', fontSize: '14px', color: '#fff' });
    muteBtn.onclick = () => {
        videoState.isMuted = !videoState.isMuted;
        muteBtn.innerText = videoState.isMuted ? 'üîá ‡πÄ‡∏õ‡∏¥‡∏î‡πÄ‡∏™‡∏µ‡∏¢‡∏á' : 'üîä ‡∏õ‡∏¥‡∏î‡πÄ‡∏™‡∏µ‡∏¢‡∏á';
        muteBtn.style.background = videoState.isMuted ? '#28a745' : '#dc3545';
        applySpeedAndMute();
    };
    panel.appendChild(muteBtn);
    
    // Add time advance button
    const timeAdvanceBtn = document.createElement('button');
    timeAdvanceBtn.innerText = '‚è≥ ‡πÄ‡∏£‡πà‡∏á‡πÄ‡∏ß‡∏•‡∏≤ x5';
    timeAdvanceBtn.style.background = '#8f00ff';
    Object.assign(timeAdvanceBtn.style, { padding: '6px 10px', border: 'none', borderRadius: '4px', cursor: 'pointer', fontSize: '14px', color: '#fff' });
    timeAdvanceBtn.onclick = () => {
        if (videoState.timeAdvanceFactor === 0) {
            videoState.timeAdvanceFactor = 5;
            timeAdvanceBtn.innerText = '‚è≥ ‡∏´‡∏¢‡∏∏‡∏î‡πÄ‡∏£‡πà‡∏á‡πÄ‡∏ß‡∏•‡∏≤';
        } else {
            videoState.timeAdvanceFactor = 0;
            timeAdvanceBtn.innerText = '‚è≥ ‡πÄ‡∏£‡πà‡∏á‡πÄ‡∏ß‡∏•‡∏≤ x5';
        }
    };
    panel.appendChild(timeAdvanceBtn);

    // Add speed buttons
    speedOptions.forEach(speed => {
        const button = document.createElement('button');
        button.innerText = `${speed}x`;
        button.style.background = '#007bff';
        Object.assign(button.style, { padding: '6px 10px', border: 'none', borderRadius: '4px', cursor: 'pointer', fontSize: '14px', color: '#fff' });
        button.onclick = () => {
            videoState.currentSpeed = speed;
            applySpeedAndMute();
        };
        panel.appendChild(button);
    });

    document.body.appendChild(panel);
    console.log('Video control panel created.');
}

// 2. Function to apply speed and mute settings to all videos
function applySpeedAndMute() {
    if (!isScriptEnabled) return;
    const videos = document.querySelectorAll('video');
    videos.forEach(video => {
        video.playbackRate = videoState.currentSpeed;
        video.muted = videoState.isMuted;
    });

    document.querySelectorAll('iframe').forEach(iframe => {
        try {
            const iframeDoc = iframe.contentDocument || iframe.contentWindow.document;
            iframeDoc.querySelectorAll('video').forEach(video => {
                video.playbackRate = videoState.currentSpeed;
                video.muted = videoState.isMuted;
            });
        } catch (e) { /* Ignore cross-origin errors */ }
    });
}

// 3. Function to smoothly advance video time
function advanceTime() {
    if (!isScriptEnabled || videoState.timeAdvanceFactor === 0) return;
    const factor = videoState.timeAdvanceFactor;

    document.querySelectorAll('video').forEach(video => {
        if (!video.paused && video.duration && video.currentTime < video.duration) {
            video.currentTime += factor;
        }
    });

    document.querySelectorAll('iframe').forEach(iframe => {
        try {
            const iframeDoc = iframe.contentDocument || iframe.contentWindow.document;
            iframeDoc.querySelectorAll('video').forEach(video => {
                if (!video.paused && video.duration && video.currentTime < video.duration) {
                    video.currentTime += factor;
                }
            });
        } catch (e) { /* Ignore cross-origin errors */ }
    });
}

// 4. Function to auto-play videos
function autoPlay() {
    if (!isScriptEnabled) return;
    document.querySelectorAll('video').forEach(video => {
        if (video.paused && video.readyState >= 2 && !video.muted) {
            video.play().catch(e => console.warn('Play error:', e));
        }
    });

    document.querySelectorAll('iframe').forEach(iframe => {
        try {
            const iframeDoc = iframe.contentDocument || iframe.contentWindow.document;
            iframeDoc.querySelectorAll('video').forEach(video => {
                if (video.paused && video.readyState >= 2 && !video.muted) {
                    video.play().catch(e => console.warn('Play error:', e));
                }
            });
        } catch (e) { /* Ignore cross-origin errors */ }
    });
}

// Create the Toggle button
const toggleBtn = document.createElement('button');
toggleBtn.innerText = 'üé¨ Script: ON';
Object.assign(toggleBtn.style, {
    position: 'fixed',
    bottom: '10px',
    right: '10px',
    zIndex: 99999,
    padding: '8px 12px',
    background: '#222',
    color: '#fff',
    border: 'none',
    borderRadius: '5px',
    cursor: 'pointer',
    fontSize: '14px'
});

toggleBtn.onclick = () => {
    isScriptEnabled = !isScriptEnabled;
    toggleBtn.innerText = `üé¨ Script: ${isScriptEnabled ? 'ON' : 'OFF'}`;
    const panel = document.getElementById('video-control-panel');
    if (isScriptEnabled) {
        if (!panel) {
            createVideoControls();
        } else {
            panel.style.display = 'flex';
        }
    } else {
        if (panel) {
            panel.style.display = 'none';
        }
    }
};

// Initial setup
document.addEventListener('DOMContentLoaded', () => {
    document.body.appendChild(toggleBtn);
});

// Main execution loop
setInterval(() => {
    if (isScriptEnabled) {
        applySpeedAndMute();
        autoPlay();
        advanceTime();
    }
}, 1000);

// Also run once to set up controls on load
if (isScriptEnabled) {
    createVideoControls();
}
