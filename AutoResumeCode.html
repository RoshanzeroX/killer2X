// ==UserScript==
// @name          VARS - Video Auto-Resume System
// @namespace     http://tampermonkey.net/
// @version       1.5
// @description   Automatically clicks the "Click to Identify" and "Confirm" buttons, and resumes videos.
// @match         https://lmsrv.siammakro.co.th/
// @grant         none
// ==/UserScript==

(function () {
    'use strict';

    let enabled = true;

    // ‡∏™‡∏£‡πâ‡∏≤‡∏á‡∏õ‡∏∏‡πà‡∏° Toggle ‡∏™‡∏≥‡∏´‡∏£‡∏±‡∏ö‡πÄ‡∏õ‡∏¥‡∏î/‡∏õ‡∏¥‡∏î‡∏™‡∏Ñ‡∏£‡∏¥‡∏õ‡∏ï‡πå
    const toggleBtn = document.createElement('button');
    toggleBtn.innerText = 'üé¨ AutoResume: ON';
    Object.assign(toggleBtn.style, {
        position: 'fixed',
        bottom: '10px',
        right: '10px',
        zIndex: 9999,
        padding: '8px 12px',
        background: '#222',
        color: '#fff',
        border: 'none',
        borderRadius: '5px',
        cursor: 'pointer',
        fontSize: '14px'
    });

    document.addEventListener('DOMContentLoaded', () => {
        document.body.appendChild(toggleBtn);
        console.log('VARS script is running and button has been added.');
    });
    
    // ‡∏ï‡∏±‡πâ‡∏á‡∏Ñ‡πà‡∏≤‡∏Å‡∏≤‡∏£‡∏Ñ‡∏•‡∏¥‡∏Å‡∏õ‡∏∏‡πà‡∏° Toggle
    toggleBtn.onclick = () => {
        enabled = !enabled;
        toggleBtn.innerText = `üé¨ AutoResume: ${enabled ? 'ON' : 'OFF'}`;
    };

    // ‡∏ü‡∏±‡∏á‡∏Å‡πå‡∏ä‡∏±‡∏ô‡∏´‡∏•‡∏±‡∏Å‡∏™‡∏≥‡∏´‡∏£‡∏±‡∏ö‡∏Ñ‡∏•‡∏¥‡∏Å‡∏õ‡∏∏‡πà‡∏°‡∏≠‡∏±‡∏ï‡πÇ‡∏ô‡∏°‡∏±‡∏ï‡∏¥
    const autoClickButtons = () => {
        if (!enabled) return;

        // ‡∏Ç‡∏±‡πâ‡∏ô‡∏ï‡∏≠‡∏ô‡∏ó‡∏µ‡πà 1: ‡∏´‡∏≤‡πÅ‡∏•‡∏∞‡∏Ñ‡∏•‡∏¥‡∏Å‡∏õ‡∏∏‡πà‡∏° "Click to Identify"
        const identifyButton = Array.from(document.querySelectorAll('button, div, a'))
            .find(el => el.textContent.trim().toLowerCase().includes('click to identify'));

        if (identifyButton && identifyButton.offsetParent !== null) {
            console.log('‚úÖ Found and clicking "Click to Identify"...');
            identifyButton.click();

            // ‡∏£‡∏≠‡∏™‡∏±‡∏Å‡∏Ñ‡∏£‡∏π‡πà‡∏Å‡πà‡∏≠‡∏ô‡∏Ñ‡∏•‡∏¥‡∏Å‡∏õ‡∏∏‡πà‡∏° "Confirm"
            setTimeout(() => {
                // ‡∏Ç‡∏±‡πâ‡∏ô‡∏ï‡∏≠‡∏ô‡∏ó‡∏µ‡πà 2: ‡∏´‡∏≤‡πÅ‡∏•‡∏∞‡∏Ñ‡∏•‡∏¥‡∏Å‡∏õ‡∏∏‡πà‡∏° "Confirm"
                const confirmButton = Array.from(document.querySelectorAll('button, div, a'))
                    .find(el => el.textContent.trim().toLowerCase().includes('confirm')); // ‡πÅ‡∏Å‡πâ‡πÑ‡∏Ç‡∏ï‡∏£‡∏á‡∏ô‡∏µ‡πâ

                if (confirmButton && confirmButton.offsetParent !== null) {
                    console.log('‚úÖ Found and clicking the Confirm button...');
                    confirmButton.click();
                }
            }, 500);
        }
    };

    setInterval(autoClickButtons, 1000);

    // ‡∏ü‡∏±‡∏á‡∏Å‡πå‡∏ä‡∏±‡∏ô‡∏™‡∏≥‡∏´‡∏£‡∏±‡∏ö‡πÄ‡∏•‡πà‡∏ô‡∏ß‡∏¥‡∏î‡∏µ‡πÇ‡∏≠‡∏ó‡∏µ‡πà‡∏´‡∏¢‡∏∏‡∏î‡πÑ‡∏ß‡πâ‡πÇ‡∏î‡∏¢‡∏≠‡∏±‡∏ï‡πÇ‡∏ô‡∏°‡∏±‡∏ï‡∏¥
    const handledVideos = new WeakSet();
    setInterval(() => {
        if (!enabled) return;
        const videos = document.querySelectorAll('video');
        videos.forEach(video => {
            if (handledVideos.has(video)) return;
            handledVideos.add(video);

            setInterval(() => {
                if (!enabled) return;
                if (video.paused && video.readyState >= 2) {
                    video.play().catch(e => console.warn('Play error:', e));
                }
            }, 1000);
        });
    }, 1000);
})();
